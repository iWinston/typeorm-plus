/**
 */

    // -------------------------------------------------------------------------
    // Public Methods
    // -------------------------------------------------------------------------
    createRepository(connection, allMetadatas, metadata) {
        return new Repository_1.Repository(connection, allMetadatas, metadata);
    }
    createTreeRepository(connection, allMetadatas, metadata) {
        return new TreeRepository_1.TreeRepository(connection, allMetadatas, metadata);
    }
    createReactiveRepository(repository) {
        return new ReactiveRepository_1.ReactiveRepository(repository);

    }
}
